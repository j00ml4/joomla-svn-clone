' FROM #__{BASENAME} AS a' . ' LEFT JOIN #__users AS co ON co.id=a.checked_out' . ' WHERE a.title LIKE ' . $database->Quote( '%' . $search . '%' ) . ' ORDER BY ' . $database->getEscaped( $orderCol ) . ' ' . ($orderDirn ? 'DESC' : 'ASC') ; $database->setQuery( $query, $limitstart, $limit ); $rows = $database->loadObjectList(); mosFS::load( '@pageNavigationAdmin' ); $total = count( $rows ); $pageNav = new mosPageNav( $total, $limitstart, $limit ); mosFS::load( '@admin_html' ); {BASENAME}Screens::view( $rows, $vars, $lists, $pageNav ); }
